  if (score >= 85) return { label: "Exceptional",         color: "#16a34a", bg: "#dcfce7" };
  if (score >= 70) return { label: "Strong",              color: "#2563eb", bg: "#dbeafe" };

  if (score >= 40) return { label: "Needs Improvement",   color: "#ea580c", bg: "#ffedd5" };
  return             { label: "At Risk",             color: "#dc2626", bg: "#fee2e2" };
}

// ─── Pentagon ──────────────────────────────────────────────────────────────
function Pentagon({ current, previous, size = 240 }) {
  const cx = size/2, cy = size/2, r = size*0.35, labelR = size*0.46;
  const offset = -Math.PI/2;
  const pt = (i, rad) => ({ x: cx + rad*Math.cos(offset + 2*Math.PI*i/5), y: cy + rad*Math.sin(offset + 2*Math.PI*i/5) });
  const path = (scores) => {
    const keys = ["delivery","quality","ai","activity","contribution"];
    return keys.map((k,i) => pt(i, (scores[k]/100)*r)).map((p,i) => `${i===0?"M":"L"}${p.x.toFixed(1)},${p.y.toFixed(1)}`).join(" ") + "Z";
  };
  const axes = Object.values(PILLAR_META).map(m => m.label);
  return (
    <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
      {[0.25,0.5,0.75,1].map((l,li) => {
        const pts = Array.from({length:5},(_,i)=>pt(i,r*l));
        return <path key={li} d={pts.map((p,i)=>`${i===0?"M":"L"}${p.x.toFixed(1)},${p.y.toFixed(1)}`).join(" ")+"Z"} fill="none" stroke="#E5E7EB" strokeWidth="1"/>;
      })}
      {Array.from({length:5},(_,i)=>{const e=pt(i,r);return <line key={i} x1={cx} y1={cy} x2={e.x} y2={e.y} stroke="#E5E7EB" strokeWidth="1"/>;}) }
      {previous && <path d={path(previous)} fill="rgba(99,102,241,0.07)" stroke="rgba(99,102,241,0.4)" strokeWidth="1.5" strokeDasharray="4 3"/>}
      <path d={path(current)} fill="rgba(37,99,235,0.10)" stroke="#2563eb" strokeWidth="2"/>
      {["delivery","quality","ai","activity","contribution"].map((k,i)=>{const p=pt(i,(current[k]/100)*r);return <circle key={k} cx={p.x} cy={p.y} r="3.5" fill="#2563eb"/>;}) }
      {axes.map((label,i)=>{const p=pt(i,labelR);const anchor=p.x<cx-5?"end":p.x>cx+5?"start":"middle";return <text key={i} x={p.x} y={p.y+4} textAnchor={anchor} fontSize="10" fill="#6B7280" fontFamily="Inter,sans-serif" fontWeight="500">{label}</text>;})}
      {previous && <>
        <circle cx={cx-44} cy={size-13} r="4" fill="rgba(99,102,241,0.4)"/>
        <text x={cx-36} y={size-9} fontSize="9" fill="#9CA3AF" fontFamily="Inter">Previous sprint</text>
        <circle cx={cx+18} cy={size-13} r="4" fill="#2563eb"/>
        <text x={cx+26} y={size-9} fontSize="9" fill="#9CA3AF" fontFamily="Inter">Current sprint</text>
      </>}
    </svg>
  );
}

// ─── Stat card ─────────────────────────────────────────────────────────────
function StatCard({ label, value, sub, bad }) {
  return (
    <div style={{background:bad?"#FEF2F2":"#F9FAFB",border:`1px solid ${bad?"#FECACA":"#E5E7EB"}`,borderRadius:8,padding:"10px 12px"}}>

      {sub && <div style={{fontSize:10,color:"#9CA3AF"}}>{sub}</div>}
    </div>
  );
}

  const [sprintFilter, setSprintFilter] = useState(SPRINTS[SPRINTS.length-1]);
  const [teamFilter,   setTeamFilter]   = useState("All");
  const [roleFilter,   setRoleFilter]   = useState("All");
  const [ratingFilter, setRatingFilter] = useState("All");
  const [sortKey,      setSortKey]      = useState("overall");
  const [sortDir,      setSortDir]      = useState("desc");

  const sprintIdx = SPRINTS.indexOf(sprintFilter);

  const rows = useMemo(() => {
    return DEVELOPERS
      .filter(d => teamFilter   === "All" || d.team === teamFilter)
      .filter(d => roleFilter   === "All" || d.role === roleFilter)
      .filter(d => {
        if (ratingFilter === "All") return true;
        return getRating(d.sprints[sprintIdx].overall).label === ratingFilter;
      })
      .map(d => ({ ...d, sd: d.sprints[sprintIdx] }))
      .sort((a,b) => {
        const dir = sortDir === "asc" ? 1 : -1;
        if (sortKey === "name")   return dir * a.name.localeCompare(b.name);
        if (sortKey === "overall") return dir * (a.sd.overall - b.sd.overall);
        return dir * ((a.sd.pillars[sortKey]||0) - (b.sd.pillars[sortKey]||0));
      });
  }, [sprintFilter, teamFilter, roleFilter, ratingFilter, sortKey, sortDir, sprintIdx]);

  const handleSort = (key) => {
    if (sortKey === key) setSortDir(d => d==="asc"?"desc":"asc");
    else { setSortKey(key); setSortDir("desc"); }
  };

  const SortIcon = ({ col }) => (
    <span style={{marginLeft:3,fontSize:9,color:sortKey===col?"#2563EB":"#D1D5DB"}}>
      {sortKey===col?(sortDir==="desc"?"▼":"▲"):"⇅"}
    </span>
  );

  const sel = { background:"#fff", border:"1px solid #E5E7EB", borderRadius:7, padding:"7px 11px", fontSize:12, color:"#374151", outline:"none", cursor:"pointer" };
  const th  = { padding:"10px 14px", fontSize:11, fontWeight:600, color:"#9CA3AF", letterSpacing:0.5, textAlign:"left", borderBottom:"1px solid #E5E7EB", background:"#F9FAFB", whiteSpace:"nowrap", cursor:"pointer", userSelect:"none" };
  const td  = { padding:"13px 14px", borderBottom:"1px solid #F3F4F6", fontSize:13, color:"#374151", verticalAlign:"middle" };

  return (
    <div style={{minHeight:"100vh",background:"#F8FAFC",fontFamily:"Inter,system-ui,sans-serif",color:"#111827"}}>
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Inter:wght@400;500;600;700;800&display=swap');
        * { box-sizing: border-box; }
        ::-webkit-scrollbar { width:5px; height:5px; }
        ::-webkit-scrollbar-track { background:#F3F4F6; }
        ::-webkit-scrollbar-thumb { background:#D1D5DB; border-radius:3px; }
        .dev-row:hover td { background:#F0F7FF !important; }
        .dev-row { cursor:pointer; }
      `}</style>

      {/* Nav */}
      <div style={{background:"#fff",borderBottom:"1px solid #E5E7EB",padding:"0 32px",display:"flex",alignItems:"center",height:54,gap:16}}>
        <div style={{display:"flex",alignItems:"center",gap:8}}>
          <div style={{width:26,height:26,background:"#2563EB",borderRadius:6,display:"flex",alignItems:"center",justifyContent:"center"}}>
            <span style={{color:"#fff",fontSize:13}}>◆</span>
          </div>
          <span style={{fontSize:14,fontWeight:700,color:"#111827"}}>DevMetrics</span>
        </div>
        <span style={{color:"#E5E7EB"}}>|</span>
        <span style={{fontSize:13,color:"#6B7280"}}>Sprint Productivity Dashboard</span>
      </div>

      <div style={{padding:"26px 32px"}}>
        <div style={{marginBottom:20}}>
          <h1 style={{margin:0,fontSize:21,fontWeight:800,color:"#111827"}}>Team Overview</h1>
          <p style={{margin:"3px 0 0",fontSize:13,color:"#9CA3AF"}}>Click any developer to view detailed sprint performance and drill into metrics</p>
        </div>

        {/* Filters */}
        <div style={{display:"flex",gap:10,marginBottom:18,flexWrap:"wrap",background:"#fff",padding:"12px 16px",borderRadius:10,border:"1px solid #E5E7EB",alignItems:"center"}}>
          {[
            { label:"SPRINT", value:sprintFilter, set:setSprintFilter, opts:SPRINTS },
            { label:"TEAM",   value:teamFilter,   set:setTeamFilter,   opts:["All",...TEAMS] },
            { label:"ROLE",   value:roleFilter,   set:setRoleFilter,   opts:["All",...ROLES] },
            { label:"RATING", value:ratingFilter, set:setRatingFilter, opts:["All",...RATING_LABELS] },
          ].map(f => (
            <div key={f.label} style={{display:"flex",alignItems:"center",gap:6}}>
              <span style={{fontSize:10,color:"#9CA3AF",fontWeight:600,letterSpacing:0.5}}>{f.label}</span>
              <select style={sel} value={f.value} onChange={e=>f.set(e.target.value)}>
                {f.opts.map(o=><option key={o}>{o}</option>)}
              </select>
            </div>
          ))}
          <div style={{marginLeft:"auto",fontSize:12,color:"#9CA3AF"}}>
            {rows.length} developer{rows.length!==1?"s":""}
          </div>
        </div>

        {/* Table */}
        <div style={{background:"#fff",borderRadius:12,border:"1px solid #E5E7EB",overflow:"hidden"}}>
          <div style={{overflowX:"auto"}}>
            <table style={{width:"100%",borderCollapse:"collapse"}}>
              <thead>
                <tr>
                  <th style={th} onClick={()=>handleSort("name")}>Developer <SortIcon col="name"/></th>
                  <th style={th}>Role</th>
                  <th style={th}>Team</th>
                  <th style={{...th,textAlign:"center"}} onClick={()=>handleSort("overall")}>Overall <SortIcon col="overall"/></th>
                  {Object.entries(PILLAR_META).map(([key,m]) => (
                    <th key={key} style={{...th,textAlign:"center"}} onClick={()=>handleSort(key)}>
                      {m.icon} {m.label} <SortIcon col={key}/>
                    </th>
                  ))}
                </tr>
              </thead>
              <tbody>
                {rows.length === 0 && (
                  <tr><td colSpan={9} style={{...td,textAlign:"center",padding:"44px",color:"#9CA3AF"}}>No developers match the current filters.</td></tr>
                )}
                {rows.map(dev => {
                  const rt = getRating(dev.sd.overall);
                  const isActive = selectedDev?.id === dev.id;
                  return (
                    <tr key={dev.id} className="dev-row" onClick={()=>setSelectedDev(isActive?null:dev)}
                      style={{background:isActive?"#EFF6FF":"transparent"}}>
                      <td style={td}>
                        <div style={{display:"flex",alignItems:"center",gap:10}}>
                          <div style={{width:34,height:34,borderRadius:"50%",background:isActive?"#BFDBFE":"#F3F4F6",color:isActive?"#1D4ED8":"#6B7280",display:"flex",alignItems:"center",justifyContent:"center",fontWeight:700,fontSize:12,flexShrink:0}}>{dev.avatar}</div>
                          <span style={{fontWeight:600,color:"#111827"}}>{dev.name}</span>
                        </div>
                      </td>
                      <td style={td}><span style={{fontSize:11,background:"#F3F4F6",color:"#6B7280",padding:"2px 8px",borderRadius:4,fontWeight:500}}>{dev.role}</span></td>
                      <td style={td}><span style={{fontSize:12,color:"#374151"}}>{dev.team}</span></td>
                      <td style={{...td,textAlign:"center"}}>
                        <div style={{display:"flex",flexDirection:"column",alignItems:"center",gap:3}}>
                          <span style={{fontSize:15,fontWeight:800,color:rt.color}}>{dev.sd.overall}</span>
                          <span style={{fontSize:10,color:rt.color,background:rt.bg,padding:"1px 7px",borderRadius:10,fontWeight:600,whiteSpace:"nowrap"}}>{rt.label}</span>
                        </div>
                      </td>
                      {Object.entries(PILLAR_META).map(([key,m]) => (
                        <td key={key} style={{...td,textAlign:"center"}}>
                          <div style={{display:"flex",flexDirection:"column",alignItems:"center",gap:3}}>
                            <PillarCell score={dev.sd.pillars[key]} color={m.color}/>
                            {key==="contribution" && dev.sd.details.contribution.items.length>0 && (
                              <span style={{fontSize:9,color:"#DB2777"}}>⭐ bonus</span>
                            )}
                          </div>
                        </td>
                      ))}
                    </tr>
                  );
                })}
              </tbody>
            </table>
          </div>
        </div>

        <div style={{marginTop:14,fontSize:11,color:"#C4C9D4",textAlign:"center"}}>
          Based on DORA · SPACE · DX Core 4 &nbsp;·&nbsp; Weights: Delivery 35% · Quality 35% · AI Leverage 15% · Activity 10% · Contribution Bonus 5% &nbsp;·&nbsp; For performance context, not sole appraisal input
        </div>
      </div>

      {/* Overlay + Panel */}
      {selectedDev && <>
        <div onClick={()=>setSelectedDev(null)} style={{position:"fixed",inset:0,background:"rgba(0,0,0,0.15)",zIndex:99}}/>
        <SidePanel dev={selectedDev} onClose={()=>setSelectedDev(null)}/>
      </>}
    </div>
  );
}
